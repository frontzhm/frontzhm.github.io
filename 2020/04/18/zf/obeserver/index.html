// 观察者和被观察者是有关联的 // 观察者需要将自己放到被观察者之上，当被观察者状态发生变化，需要通知所有的观察者 // 宝宝和父母，宝宝就是被观察者 父母就是观察者 class Subject{ constructor(name){ this.name = name this.state = '开心' this.observers = [] } // 将观察者放进数组，也就是愿意让这个数组里面的观察者 观察 attach(observer){ // 相当于发布订阅的订阅on，收集observer this.observers.push(observer) } // 当被观察者的状态发生变化的时候，让每个观察者都知道这个事，观察者会根据状态做出相应行为 setState(state){ this.state = state // 相当于发布订阅的发布emit，让observer都行动起来 this.observers.forEach(o=>{ o.update(this) }) } } class Observer{ constructor(name){ this.name = name } // 只要被观察者的状态发生变化就会调用这个方法 update(s){ console.log(`${this.name}发现 ${s.name}${s.state}`) } } let baby = new Subject('宝宝') let baba = new Observer('爸爸') let mama = new Observer('妈妈') baby.attach(baba) baby.attach(mama) baby.setState('不开心') baby.setState('开心')
